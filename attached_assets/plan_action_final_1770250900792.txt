2. VÉRIFIER LE SERVEUR EXPRESS (CRITIQUE)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   ChatGPT a identifié un problème probable : votre serveur écoute peut-être sur 127.0.0.1 au lieu de 0.0.0.0
   
   Vérifiez votre fichier server.js ou index.js (Express) :
   
   ```javascript
   // ❌ MAUVAIS - Ne fonctionne pas sur Replit mobile
   app.listen(3000, 'localhost', () => {
     console.log('Server running');
   });
   
   // ❌ MAUVAIS - Ne fonctionne pas sur Replit mobile
   app.listen(3000, '127.0.0.1', () => {
     console.log('Server running');
   });
   
   // ✅ BON - Fonctionne partout
   app.listen(3000, '0.0.0.0', () => {
     console.log('Server running on 0.0.0.0:3000');
   });
   
   // ✅ ENCORE MIEUX - Utilise le port de Replit
   const PORT = process.env.PORT || 3000;
   app.listen(PORT, '0.0.0.0', () => {
     console.log(`Server running on 0.0.0.0:${PORT}`);
   });
   ```
   
   🎯 CORRECTION IMMÉDIATE :
   
   Remplacez votre app.listen() par :
   ```javascript
   const PORT = process.env.PORT || 3000;
   const HOST = '0.0.0.0'; // CRUCIAL pour Replit
   
   app.listen(PORT, HOST, () => {
     console.log(`✅ Server listening on ${HOST}:${PORT}`);
     console.log(`🌍 Public URL: ${process.env.REPL_SLUG}.${process.env.REPL_OWNER}.repl.co`);
   });
   ```

3. AJOUTER UN SCRIPT DE DIAGNOSTIC D'URGENCE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   Dans votre index.html (celui de Vite), ajoutez TOUT EN HAUT du <head> :
   
   ```html
   <!DOCTYPE html>
   <html lang="fr">
   <head>
     <meta charset="UTF-8">
     <meta name="viewport" content="width=device-width, initial-scale=1.0">
     
     <!-- DIAGNOSTIC D'URGENCE - À SUPPRIMER UNE FOIS LE BUG RÉSOLU -->
     <script>
       window.debugStartTime = Date.now();
       window.debugErrors = [];
       
       // Capturer TOUTES les erreurs
       window.addEventListener('error', function(e) {
         window.debugErrors.push({
           type: 'error',
           message: e.message,
           source: e.filename,
           line: e.lineno,
           time: Date.now() - window.debugStartTime
         });
         console.error('❌ ERROR:', e.message, 'at', e.filename + ':' + e.lineno);
       }, true);
       
       // Capturer les promesses rejetées
       window.addEventListener('unhandledrejection', function(e) {
         window.debugErrors.push({
           type: 'promise',
           reason: String(e.reason),
           time: Date.now() - window.debugStartTime
         });
         console.error('❌ PROMISE REJECTION:', e.reason);
       });
       
       // Afficher un diagnostic après 2 secondes si rien ne s'affiche
       setTimeout(function() {
         var bodyHeight = document.body?.offsetHeight || 0;
         var bodyText = document.body?.innerText || '';
         
         console.log('=== DIAGNOSTIC ===');
         console.log('Time elapsed:', Date.now() - window.debugStartTime + 'ms');
         console.log('Body height:', bodyHeight);
         console.log('Body text length:', bodyText.length);
         console.log('Errors:', window.debugErrors);
         
         if (bodyHeight < 100 || bodyText.length < 20) {
           console.error('🚨 EMERGENCY: Content not visible!');
           
           // Afficher un écran d'erreur informatif
           document.body.innerHTML = '<div style="padding:30px;font-family:sans-serif;background:#fee;color:#800;"><h1>🚨 Erreur de chargement mobile</h1><h2>Informations de diagnostic :</h2><pre style="background:#fff;padding:15px;overflow:auto;font-size:12px;">' + JSON.stringify({
             userAgent: navigator.userAgent,
             viewport: window.innerWidth + 'x' + window.innerHeight,
             screen: screen.width + 'x' + screen.height,
             timeElapsed: Date.now() - window.debugStartTime + 'ms',
             errors: window.debugErrors,
             bodyHeight: bodyHeight,
             textLength: bodyText.length
           }, null, 2) + '</pre><button onclick="location.reload()" style="padding:12px 24px;background:#3b82f6;color:#fff;border:none;border-radius:8px;font-size:16px;margin-top:20px;cursor:pointer;">Recharger</button><button onclick="location.href=\'/test.html\'" style="padding:12px 24px;background:#10b981;color:#fff;border:none;border-radius:8px;font-size:16px;margin-top:20px;margin-left:10px;cursor:pointer;">Page de test</button></div>';
         } else {
           console.log('✅ Content appears to be visible');
         }
       }, 2000);
       
       console.log('🔍 Mobile debugger initialized');
     </script>
     
     <title>Sites web premium — done</title>
     <!-- Reste de votre head -->
   </head>
   <body>
     <!-- Fallback visible pendant le chargement -->
     <div id="root">
       <div style="display:flex;align-items:center;justify-content:center;min-height:100vh;font-family:sans-serif;color:#666;">
         Chargement...
       </div>
     </div>
     <!-- Vos scripts -->
   </body>
   </html>
   ```
